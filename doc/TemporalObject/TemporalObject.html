def add_timeline(timeline)
  raise ArgumentError, "Can only add a TimeSpan::TimeLine to a TemporalObject's statuses" unless timeline.kind_of? TimeSpan::TimeLine
  @statuses << timeline
end

  @statuses.delete(timeline)
